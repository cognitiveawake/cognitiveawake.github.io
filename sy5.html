2. 使用jQuery，以交互方式浏览一组图片


<html>
    <head>
        <meta charset="utf-8">
        <title>以交互方式显示一组图片</title>
        <script type="text/javascript" src="js/jquery-1.3.2.js"></script>
        <!--样式定义略-->
    </head>
    <body>
        <div class="big">
            <div class="row1"><img src="images/p1.jpg" width="300" height="170" name="tpk"></div>
            <div class="row2">
                <input type="button" value="上一幅" id="rf">
                共有<span id="msg1"></span>幅&nbsp;&nbsp;当前序号：<span id="msg2"></span>
                <input type="button" value="下一幅" id="ff"></div>
        </div>
    </body>
    <script>
        $(document).ready(function(){
            var tp=new Array(5);tp[0]="images/p1.jpg";
            tp[1]="images/p2.jpg";tp[2]="images/p3.jpg";tp[3]="images/p4.jpg";tp[4]="images/p5.jpg";
            var next=$("#ff"); //通过id样式获取按钮对象
            var last=$("#rf"); 
            var index=0;  
            $("#msg1"). html (tp.length); //设置指定位置的文本
            setButtonState();
            function setButtonState(){  //定义状态函数（方法）
                index==0?last.css("display", "none"):last.css("display","inline");  
                index==tp.length?next.css("display", "none"):next.css("display","inline"); 
                $("#msg2").html(index+1);
        }
            next.click(function(){
                index++;
                tpk.src=tp[index];          
                setButtonState();  //调用函数更新状态
            });
            last.click(function(){
                index--;
                tpk.src=tp[index];
                setButtonState();
            });
    });
    </script>
</html>

4. jQuery处理表单元素——会员登录页面

页面的客户端脚本（JS代码）：

<script>
    $(document).ready(function () {
        var users = [["wzx", "123"], ["wyh", "456"]];  //二维数组即是数组中的每个元素是一个数组，初始化2个用户
        users.push(["wj", "789"]);  //追加一个用户信息

        $("#ensure").click(function () {
            un = $("#username").val(); //获取文本框的输入值
            pwd = $("#password").val();  //获取密码框的输入值
            for (i = 0; i < users.length; i++) {
                if (users[i][0] == un) break;
            }
            if (i < users.length) {
                if (users[i][1] == pwd) {  //登录成功
                    alert("登录成功！");
                    $("#state").html("欢迎你：" + un);  //局部刷新
                    $("table").hide("slow");  //隐藏表格
                } else {
                    alert("密码错误...");
                }
            } else {
                alert("用户名不存在...");
            }
        });

        $("#cancel").click(function () {  //响应取消按钮
            location.href = "index.html";
        });
    });
</script>

代码要点如下：

  • 一维数组初始化时使用一对中括号
  • 数组对象使用push()方法来添加元素
  • jQuery对象具有隐藏方法hide()
  • 遍历jQuery对象（元素）集的方法each()
  • 使用浏览器二级对象的方法实现页面跳转 location.href
  • 使用jQuery对象的html()方法实现页面局部刷新
